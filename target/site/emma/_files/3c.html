405   /**
406    * Raise self to an integer exponent.
407    * @param exponent The exponent.
408    */
409   public void powSelf(long exponent)
410   {
411      if (isNaN())
412         return;
413      if (exponent == 0)
414      {
415         if (myHigh == 0)
416         {
417            // 0^0 = NaN
418            myHigh = Double.NaN;
419            myLow = 0;
420         }
421         else
422         {
423            // x^0 = 1
424            myHigh = 1;
425            myLow = 0;
426         }
427      }
428      // 0^y = 0, 1^y = 1.
429      if (myHigh == 0 || (myHigh == 1 && myLow == 0))
430         return;
431      
432      boolean invert = exponent < 0;
433      exponent = Math.abs(exponent);
434 
435      if (exponent == 2)
436         squareSelf();
437      else if (exponent > 2)
438      {
439         // Exponentiation by repeated squaring.
440         DoubleDouble result = new DoubleDouble(1.0);
441         DoubleDouble square = new DoubleDouble(this);
442         while(exponent >= 1)
443         {
444            if ((exponent & 1) == 1) // if odd
445               result.multiplySelfBy(square);
446            square.squareSelf();
447            exponent >>>= 1;  // int divide by 2
448         }
449         myHigh = result.myHigh;
450         myLow = result.myLow;
451      }
452 
453      if (invert)
454      {
455         DoubleDouble reciprocal = new DoubleDouble(1.0);
456         reciprocal.divideSelfBy(this);
457         myHigh = reciprocal.myHigh;
458         myLow = reciprocal.myLow;
459      }
460   }

485   /**
486    * Normalize this <code>Double</code> following an arithmetic computation.
487    * @param s0 The high order term.
488    * @param s1 The low order term.
489    * @param e The error.
490    */
491   private void normalize(double s0, double s1, double e)
492   {
493      double t0, t1, t2, v;
494      int k = 0;
495      // Normalize: "a0" is s0, "a1" is s1, "a2" is e.
496      // Quick Two Sum: "a" is s1, "b" is e, "s" is v (reused variable), "e" is t2.
497      v = s1 + e;
498      t2 = e - (v - s1);
499      // Quick Two Sum: "a" is s0, "b" is v, "s" is t0, "e" is t1.
500      t0 = s0 + v;
501      t1 = v - (t0 - s0);
502 
503      myHigh = myLow = 0;
504      // Quick Two Sum: "a" is t0, "b" is t1, "s" is v (reused variable), "e" is e.
505      v = t0 + t1;
506      e = t1 - (v - t0);
507 
508      if (v != 0)
509      {
510         myHigh = v;
511         myLow = 0;
512         v = e;
513         k++;
514      }
515      // Quick Two Sum: "a" is v, "b" is t2, "s" is s0 (reused variable), "e" is e.
516      s0 = v + t2;
517      // Calculation of e only if needed later (in myLow if k == 0).
518      if (s0 != 0)
519      {
520         if (k == 0)
521         {
522            myHigh = s0;
523            myLow = t2 - (s0 - v);
524         }
525         else
526         {
527            myLow = s0;
528         }
529      }
530   }
