347                  currAgg.setInUse(false);
348                  prevValue.setAggregateValue(prevAgg, prevAgg.terminate());
349                  if (!useSuperAggregation)
350                     prevAgg.setInUse(false);
351               }
352               if (properties != null)
353                  prevValue.assignPropsAndGroupingSet(properties, defaultGroupingSet);
354               if (useSuperAggregation)
355                  prevValue.assignAggregators(prevAggsList);
356               aggValues.add(prevValue);
357            }
358            else
359            {
360               // If no match, then terminate both previous and current initial.
361               AggregateValue<T> prevValue = new AggregateValue<T>(prevObject);
362               AggregateValue<T> currValue = new AggregateValue<T>(currObject);
363               for (int a = 0; a < aggListSize; a++)
364               {
365                  Aggregator prevAgg = prevAggsList.get(a);
366                  prevValue.setAggregateValue(prevAgg, prevAgg.terminate());
367                  Aggregator currAgg = currAggsList.get(a);
368                  currValue.setAggregateValue(currAgg, currAgg.terminate());
369                  if (properties != null)
370                  {
371                     prevValue.assignPropsAndGroupingSet(properties, defaultGroupingSet);
372                     currValue.assignPropsAndGroupingSet(properties, defaultGroupingSet);
373                  }
374                  if (!useSuperAggregation)
375                  {
376                     prevAgg.setInUse(false);
377                     currAgg.setInUse(false);
378                  }
379               }
380               if (useSuperAggregation)
381               {
382                  prevValue.assignAggregators(prevAggsList);
383                  currValue.assignAggregators(currAggsList);
384               }
385               aggValues.add(prevValue);
386               aggValues.add(currValue);
387            }
388            // Add any current middles.
389            aggValues.addAll(curr.getMiddleAggValues());
390            // Make the current ending the previous.
391            prevObject = curr.getEndingObject();
392            prevAggsList = curr.getEndingAggList();
393         }
394      }
395      // 3. Off the end of the List.  Wrap up the last one.
396      // In any above case, the last one is "previous".
397      // Terminate it.
398      AggregateValue<T> prevValue = new AggregateValue<T>(prevObject);
399      for (int a = 0; a < aggListSize; a++)
400      {
401         Aggregator prevAgg = prevAggsList.get(a);
402         prevValue.setAggregateValue(prevAgg, prevAgg.terminate());
403         if (!useSuperAggregation)
404            prevAgg.setInUse(false);
405      }
406      if (properties != null)
407         prevValue.assignPropsAndGroupingSet(properties, defaultGroupingSet);
408      if (useSuperAggregation)
409         prevValue.assignAggregators(prevAggsList);
410      aggValues.add(prevValue);
411 
412      return aggValues;
413   }

415   /**
416    * In the already sorted list, return the highest index whose item in the
417    * list compares equal to the item at the given start index.
418    *
419    * @param <T> The type of objects in the <code>List</code> of values.
420    * @param list The <code>List</code> of values.
421    * @param comparator Decides how to compare values for equality.
422    * @param startIdx Start looking for the last match at this index.
423    * @return The index that represents the last object in the given
424    *    <code>List</code> that compares equal to the object represented by
425    *    the start index.
426    */
427   public static <T> int indexOfLastMatching(List<T> list,
428      Comparator<? super T> comparator, int startIdx)
429   {
430      return indexOfLastMatching(list, comparator, startIdx, list.size() - 1);
431   }
432 
433   /**
434    * In the already sorted list, return the highest index whose item in the
435    * list compares equal to the item at the given start index, except that no
436    * value larger than the maximum index will be returned.
437    * @param <T> The type of objects in the <code>List</code> of values.
438    * @param list The <code>List</code> of values.
439    * @param comparator Decides how to compare values for equality.
440    * @param startIdx Start looking for the last match at this index.
441    * @param maxIdx Don't look past this index.
442    * @return The lesser of the index that represents the last object in the
443    *    given <code>List</code> that compares equal to the object represented
444    *    by the start index, and <code>maxIdx</code>.
445    */
446   public static <T> int indexOfLastMatching(List<T> list,
447      Comparator<? super T> comparator, int startIdx, int maxIdx)
448   {
449      T value = list.get(startIdx);
450      int addMatchIdx = 1;
451      // lowerBoundMatchIdx is what will be returned.
452      int lowerBoundMatchIdx = startIdx;
453      int upperBoundMatchIdx = startIdx + addMatchIdx;
454 
455      // Find out lower/upper bound of last match by repeatedly doubling
456      // addMatchIdx until the end of the list is reached or we have found an
457      // item that doesn't match.
458      while (true)
459      {
460         // Don't go off the end of the List.
461         if (upperBoundMatchIdx >= maxIdx)
462         {
463            upperBoundMatchIdx = maxIdx;
464            break;
465         }
466 
467         if (comparator.compare(value, list.get(upperBoundMatchIdx)) == 0)
468         {
469            // The item at upperBoundMatchIdx matches.  Adjust lower/upper
470            // bounds.
471            lowerBoundMatchIdx = upperBoundMatchIdx;
472            //addMatchIdx *= 2;
473            addMatchIdx <<= 1;
474            upperBoundMatchIdx += addMatchIdx;
475         }
476         else
477         {
478            // The item at startIdx + addMatchIdx doesn't match.  Lower/upper
479            // bounds are correct.
480            break;
481         }
482      }
483 
484      // Binary search between lower/upper bound to find the last item.
485      while (true)
486      {
487         //int midMatchIdx = (lowerBoundMatchIdx + upperBoundMatchIdx) / 2;
488         int midMatchIdx = (lowerBoundMatchIdx + upperBoundMatchIdx) >> 1;
489 
490         // If we have a match, then we are done.
491         if (lowerBoundMatchIdx == upperBoundMatchIdx) break;
492         // If two possibilities, left, special logic to determine which one:
493         // lower or upper.  It's necessary because otherwise an infinite loop
494         // would result where midMatchIdx continuously evaluates to
495         // lowerBoundMatchIdx.
496         boolean downToTwo = (lowerBoundMatchIdx == upperBoundMatchIdx - 1);
497 
498         if (comparator.compare(value, list.get(midMatchIdx)) == 0)
499         {
500            if (downToTwo)
501            {
502               // lowerBoundMatchIdx == midMatchIdx
503               if (comparator.compare(value, list.get(upperBoundMatchIdx)) == 0)
504               {
505                  // Upper bound compares equal also.  It's the upper bound.
506                  lowerBoundMatchIdx = upperBoundMatchIdx;
507               }
508               // Else, upper bound doesn't compare equal.  It's the lower bound.
509               // Either way, we've found our index.  Break.
510               break;
511            }
512            // The item at midMatchIdx matches.  Set lower bound here.
513            lowerBoundMatchIdx = midMatchIdx;
514         }
515         else
516         {
517            // The item at midMatchIdx doesn't match.  Set upper bound at here - 1.
518            // Note that in this case it is impossible to be down to two.
519            // Otherwise at least the lower bound would have matched.
520            upperBoundMatchIdx = midMatchIdx - 1;
521         }
522      }
523 
524      return lowerBoundMatchIdx;
525   }
