363    * super-aggregate values are added to the given <code>List</code> of
364    * <code>AggregateValues</code>.  It is expected that each
365    * <code>AggregateValue</code> has been assigned a reference to the
366    * <code>List</code> of <code>Aggregators</code> that will be used to
367    * calculate super-aggregate results.  They are released after the
368    * calculations are complete.
369    * @param aggValues A <code>List</code> of <code>AggregateValues</code>.
370    * @param <T> The type of object to super-aggregate.
371    */
372   private <T> void getSuperAggregateValues(List<AggregateValue<T>> aggValues)
373   {
374      if (DEBUG)
375      {
376         System.err.println("A.gSAV begin");
377         System.err.println("Grouping Sets:");
378         for (List<Integer> groupingSet : myGroupingSets)
379         {
380            System.err.println(groupingSet);
381         }
382      }
383      boolean includeOrigAggValues = false;
384      // Define the chain of grouping sets, which traces its origin back to the
385      // original aggValues passed in (-1).
386      Map<Integer, Integer> chainedGroupingSets = new HashMap<Integer, Integer>();
387      int numGroupingSets = myGroupingSets.size();
388      int numProperties = myProperties.size();
389      
390      for (int i = 0; i < numGroupingSets; i++)
391      {
392         List<Integer> groupingSet = myGroupingSets.get(i);
393         int j;
394         for (j = i - 1; j >= 0; j--)
395         {
396            List<Integer> candidateSet = myGroupingSets.get(j);
397            if (candidateSet.containsAll(groupingSet))
398            {
399               chainedGroupingSets.put(i, j);
400               if (DEBUG)
401                  System.err.println("  put(" + i + ", " + j + ")");
402               break;
403            }
404         }
405         // Must rely on original agg values.
406         if (j == -1)
407         {
408            if (DEBUG)
409               System.err.println("  put(" + i + ", -1)");
410            chainedGroupingSets.put(i, -1);
411         }
412      }
413      int start = 0;
414      if (myGroupingSets.get(0).size() == numProperties)
415      {
416         // The grouping set consisting of all properties is found.
417         // Include the original agg values in the final result set.
418         includeOrigAggValues = true;
419         start = 1;
420      }
421      if (DEBUG)
422         System.err.println("includeOrigAggValues is " + includeOrigAggValues);
423      // Walk through the grouping sets, using a prior grouping set's
424      // super-aggregate values to create the current grouping set's
425      // super-aggregate values.
426      Map<Integer, List<AggregateValue<T>>> aggValuesByGroupingSet = new TreeMap<Integer, List<AggregateValue<T>>>();
427      if (includeOrigAggValues)
428      {
429         aggValuesByGroupingSet.put(0, aggValues);
430      }
431      for (int g = start; g < numGroupingSets; g++)
432      {
433         int prevIndex = chainedGroupingSets.get(g);
434         if (DEBUG)
435            System.err.println("  get(" + g + ") = " + prevIndex);
436         List<AggregateValue<T>> useValues;
437         if (prevIndex == -1)
438            useValues = aggValues;
439         else
440            useValues = aggValuesByGroupingSet.get(prevIndex);
441 
442         // Get list of properties to super-aggregate by and sort the aggregate values.
443         List<Integer> groupingSet = myGroupingSets.get(g);
444         List<String> properties = new ArrayList<String>();
445         for (int propIndex : groupingSet)
446         {
447            properties.add(myProperties.get(propIndex));
448         }
449         Comparator<AggregateValue<T>> comparator = new AggregateValuePropertiesComparator<T>(properties);
450         if (DEBUG)
451            System.err.println("  Sorting on " + properties.toString());
452         Collections.sort(useValues, comparator);
453 
454         if (DEBUG)
455         {
456            System.err.println("  useValues: ");
457            for (AggregateValue<T> useValue : useValues)
458            {
459               System.err.println(useValue);
460            }
461         }
462 
463         List<AggregateValue<T>> superAggValues = new ArrayList<AggregateValue<T>>();
464         int aggSize = myAggregators.size();
465         int startIndex = 0;
466         int endIndex;
467         int listsize = useValues.size();
468         while (startIndex < listsize)
469         {
470            if (DEBUG)
471               System.err.println("    startIndex is " + startIndex);
472            AggregateValue<T> superAggValue = new AggregateValue<T>(useValues.get(startIndex));
473            endIndex = Aggregations.indexOfLastMatching(useValues, comparator, startIndex);
474            if (DEBUG)
475               System.err.println("    endIndex is " + endIndex);
476 
477            List<Aggregator> superAggList = getAggregatorsList();
478 
479            for (int a = 0; a < aggSize; a++)
480            {
481               Aggregator agg = superAggList.get(a);
482               agg.init();
483            }
484            for (int i = startIndex; i <= endIndex; i++)
485            {
486               AggregateValue<T> baseValue = useValues.get(i);
487               List<Aggregator> baseAggs = baseValue.retrieveAggregators();
488               for (int a = 0; a < aggSize; a++)
489               {
490                  Aggregator agg = superAggList.get(a);
491                  agg.merge(baseAggs.get(a));
492               }
493            }
494            for (int a = 0; a < aggSize; a++)
495            {
496               Aggregator agg = superAggList.get(a);
497               superAggValue.setAggregateValue(agg, agg.terminate());
498            }
499            superAggValue.assignAggregators(superAggList);
500            superAggValue.assignPropsAndGroupingSet(myProperties, groupingSet);
501            superAggValues.add(superAggValue);
502 
503            // Setup for next loop (if any).
504            startIndex = endIndex + 1;
505         }
506         if (DEBUG)
507         {
508            System.err.println("  superAggValues created for grouping set # " + g);
509            for (AggregateValue<T> superAggValue : superAggValues)
510            {
511               System.err.println(superAggValue);
512            }
513         }
514         aggValuesByGroupingSet.put(g, superAggValues);
515      }
516      // Super-aggregation is done.
517 
518      // Cleanup all the assigned Aggregators.
519      for (AggregateValue aggValue : aggValues)
520      {
521         aggValue.releaseAggregators();
522      }
523      for (int groupingSetIndex : aggValuesByGroupingSet.keySet())
524      {
525         for (AggregateValue superAggValue : aggValuesByGroupingSet.get(groupingSetIndex))
526         {
527            superAggValue.releaseAggregators();
528         }
529      }
530 
531      // Manipulate the passed in aggregate values to reflect the proper
532      // aggregates (if any) and all super-aggregates.
533      if (!includeOrigAggValues)
534      {
535         aggValues.clear();
536      }
537 
538      // Add in all super-aggregate values.
539      for (int groupingSetIndex : aggValuesByGroupingSet.keySet())
540      {
541         // Don't include orig agg values again!
542         if (groupingSetIndex == 0 && includeOrigAggValues)
543            continue;
544 
545         List<AggregateValue<T>> groupAggValues = aggValuesByGroupingSet.get(groupingSetIndex);
546 
547         if (DEBUG)
548         {
549            System.err.println("For groupingSetIndex " + groupingSetIndex +
550               ", adding ");
551            for (AggregateValue<T> groupAggValue : groupAggValues)
552            {
553               System.err.println(groupAggValue);
554            }
555         }
556 
557         aggValues.addAll(groupAggValues);
558      }
559   }

784       * property references not listed ("2, 4") exist in every grouping set.</p>

787       * <p>Because <code>cube(x, y)</code> is the same as
788       * <code>cube(x), cube(y)</code>, this method only accepts a simple list,
789       * instead of a list of lists.
790       * @param cube A <code>List</code> of integer references.  Grouping
791       *    sets are created that each contain all properties not referenced
792       *    here, and each individual grouping set contains a different number
793       *    of referenced properties <em>in order</em>.
794       * @return This <code>Builder</code>.

796       *    once; if any index is out of range from 0 to <em>n</em> -1, where
797       *    <em>n</em> is the number of "group by" properties.
798       * @see #setGroupingSets
799       * @see #setProperties
800       */

827      /**
828       * Add all rollup grouping sets to the given list of grouping sets.
829       * Initially, it is expected that the first time this is called, that
830       * <code>groupingSets</code> contains only the default grouping set.
831       * This method walks through all rollup combinations by processing the
832       * first rollup set and calling itself via recursion with the rest of the
833       * rollups list.
834       * @param groupingSets The <code>List</code> of grouping sets.  This is
835       *    modified by adding all grouping sets found through rollup
836       *    combinations.
837       * @param rollups The <code>List</code> of rollup sets.
838       */
839      private void findRollups(List<List<Integer>> groupingSets, List<List<Integer>> rollups)
840      {
841         // Ending condition of recursion.
842         if (rollups.isEmpty())
843            return;
844         if (DEBUG)
845            System.err.println("A.findRollups");
846         // Process the first rollup in the list.
847         List<Integer> rollup = rollups.get(0);
848         int rollupSize = rollup.size();
849         if (DEBUG)
850            System.err.println("  rollup: " + rollup);
851 
852         List<List<Integer>> newGroupingSets = new ArrayList<List<Integer>>();
853         for (int i = rollupSize - 1; i >= 0; i--)
854         {
855            List<Integer> remove = rollup.subList(i, rollupSize);
856            for (List<Integer> groupingSet : groupingSets)
857            {
858               List<Integer> newGroupingSet = new ArrayList<Integer>(groupingSet);
859               newGroupingSet.removeAll(remove);
860               if (DEBUG)
861                  System.err.println("  i: " + i + " add: " + newGroupingSet);
862               newGroupingSets.add(newGroupingSet);
863            }
864         }
865         groupingSets.addAll(newGroupingSets);
866         findRollups(groupingSets, rollups.subList(1, rollups.size()));
867      }

869      /**
870       * Add all cube grouping sets to the given list of grouping sets.
871       * Initially, it is expected that the first time this is called, that
872       * <code>groupingSets</code> contains only the default grouping set.
873       * @param groupingSets The <code>List</code> of grouping sets.  This is
874       *    modified by adding all grouping sets found through rollup
875       *    combinations.
876       * @param cube The <code>List</code> of cube integer references.
877       */
878      private void findCubes(List<List<Integer>> groupingSets, List<Integer> cube)
879      {
880         if (cube.isEmpty())
881            return;
882         if (DEBUG)
883            System.err.println("A.findCubes: " + cube);
884 
885         // Note: This limits cubes to a maximum of 32 properties.
886         int numCubes = cube.size();
887         int numCombinations = 1 << numCubes;
888 
889         List<Integer> defGroupingSet = groupingSets.get(0);
890         List<Integer> remove = new ArrayList<Integer>();
891         // Don't add the base grouping set (all props) again!
892         for (int i = 1; i < numCombinations; i++)
893         {
894            remove.clear();
895            for (int b = 0, bit = 1; b < numCubes; b++, bit <<= 1)
896            {
897               if ((i & bit) > 0)
898                  remove.add(cube.get(b));
899            }
900            List<Integer> combo = new ArrayList<Integer>(defGroupingSet);
901            combo.removeAll(remove);
902            if (DEBUG)
903               System.err.println("Adding cube grouping set: " + combo);
904            groupingSets.add(combo);
905         }
906      }

924      /**
925       * Validates that all integer references are within range and that no
926       * reference is repeated, even if it is repeated across special sets.
927       * This validation is intended for cube sets and rollup sets.
928       * @param specials A <code>List</code> of <code>Lists</code> of integer
929       *    references to properties, means for "special" sets such as cubes
930       *    and rollups.
931       */
932      private void validateSpecialSets(List<List<Integer>> specials)
933      {
934         int numProperties = myProperties.size();
935         // Note: This restricts grouping sets to maximum 32 fields.
936         int fieldMask = 0;
937         for (List<Integer> special : specials)
938         {
939            for (int field : special)
940            {
941               if (field < 0 || field >= numProperties)
942                  throw new IllegalArgumentException("Grouping set field index out of range: " + field);
943               int bit = 1 << field;
944               if ((fieldMask & bit) != 0)
945                  throw new IllegalArgumentException("Can't specify same field more than once in any cube/rollup set: " + field);
946               fieldMask |= bit;
947            }
948         }
949      }

951      /**
952       * Validates the grouping sets.
953       * @param groupingSets A <code>List</code> of grouping sets, which are
954       *    <code>Lists</code> of integers.
955       * @return Another <code>List</code> of grouping sets, sorted in a
956       *    particular order: <code>Lists</code> are sorted by length
957       *    descending, and each individual <code>List</code> has its integer
958       *    field references sorted in ascending order.
959       * @throws IllegalArgumentException If any index is outside the range
960       *    from 0 to <em>n</em> -1, where <em>n</em> is the number of "group
961       *    by" properties; if any index is repeated within the same grouping
962       *    set; if any grouping set is a duplicate of any other, even if the
963       *    fields are in a different order.
964       */
965      private List<List<Integer>> validateGroupingSets(List<List<Integer>> groupingSets)
966      {
967         int numProperties = myProperties.size();
968         List<List<Integer>> groupingSetsCopy = new ArrayList<List<Integer>>(groupingSets);
969         for (List<Integer> groupingSet : groupingSetsCopy)
970         {
971            if (groupingSet.size() != numProperties)
972               amIUsingSuperAggregation = true;
973 
974            // Note: This restricts grouping sets to maximum 32 fields.
975            int fieldMask = 0;
976            for (int field : groupingSet)
977            {
978               if (field < 0 || field >= numProperties)
979                  throw new IllegalArgumentException("Grouping set field index out of range: " + field);
980               int bit = 1 << field;
981               if ((fieldMask & bit) != 0)
982                  throw new IllegalArgumentException("Can't specify same field more than once in a grouping set: " + field);
983               fieldMask |= bit;
984            }
985            // Don't allow duplicate grouping sets.
986            // Check for same fields in any order.
987            for (List<Integer> otherGroupingSet : groupingSetsCopy)
988            {
989               // Only check the previous sets versus this one.
990               if (otherGroupingSet == groupingSet)
991               {
992                  break;
993               }
994               // Different size, different sets.
995               if (otherGroupingSet.size() != groupingSet.size())
996               {
997                  continue;
998               }
999               Set<Integer> fields = new HashSet<Integer>(groupingSet);
1000               for (int otherField : otherGroupingSet)
1001               {
1002                  // Does this set contain the element of the other set?
1003                  // If not, then they aren't equal.
1004                  if (!fields.remove(otherField))
1005                     break;
1006               }
1007               if (fields.isEmpty())
1008               {
1009                  // Sets with the same contents, even if in different orders,
1010                  // will reach here.
1011                  throw new IllegalArgumentException("Duplicate grouping sets found: " +
1012                     groupingSet.toString() + " and " + otherGroupingSet.toString());
1013               }
1014            }
1015            // Sort each individual list.
1016            Collections.sort(groupingSet);
1017         }
1018         // Sort the list of lists according to a specific order.
1019         Collections.sort(groupingSetsCopy, new Comparator<List<Integer>>() {
1020            /**
1021             * Sort by length of list descending.
1022             * @param list1 A <code>List</code> of integers.
1023             * @param list2 A <code>List</code> of integers.
1024             * @return An integer less than zero, equal to zero, or greater than
1025             *    zero, depending on whether the first list is longer, as long
1026             *    as, or shorter than the second list.
1027             */
1028            public int compare(List<Integer> list1, List<Integer> list2)
1029            {
1030               return list2.size() - list1.size();
1031               // The specific elements matter here, e.g. {0, 1} vs. {0, 2} vs.
1032               // {1, 2} do not matter.
1033            }
1034         });
1035         return groupingSetsCopy;
1036      }
